juce_add_plugin(PitchTests
    PRODUCT_NAME "Pitch Tests"
    MICROPHONE_PERMISSION_ENABLED   TRUE
    FORMATS Standalone VST3)

juce_generate_juce_header(PitchTests)

set(ApplicationSources
    ${UISources}
    ${CMAKE_SOURCE_DIR}/playground/pitch/PitchDetectionProcessor.h
    ${CMAKE_SOURCE_DIR}/playground/pitch/PitchDetectionProcessor.cpp
)
source_group("Source/ApplicationSources" FILES ${ApplicationSources})

target_sources(PitchTests PRIVATE
    ${ApplicationSources}
)

target_compile_definitions(PitchTests PRIVATE
    JUCE_APPLICATION_NAME_STRING="$<TARGET_PROPERTY:AudioDecay,JUCE_PRODUCT_NAME>"
    JUCE_APPLICATION_VERSION_STRING="$<TARGET_PROPERTY:AudioDecay,JUCE_VERSION>")

target_link_libraries(PitchTests 
    PRIVATE 
    aubio
    Shared_VST_Target)
